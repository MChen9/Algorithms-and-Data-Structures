class Solution {
    public final static String[] MORSE = 
    {".-", "-...","-.-.","-..",".","..-.","--.","....","..",
     ".---","-.-",".-..","--","-.","---",".--.","--.-",".-.",
     "...","-","..-","...-",".--","-..-","-.--","--.."};
    
    public int uniqueMorseRepresentations(String[] words) {
        Set<String> set = new HashSet<>();
        StringBuilder sb;
        for (String word: words) {
            sb = new StringBuilder();
            for (char c: word.toCharArray()) {
                sb.append(MORSE[c - 'a']);
            }
            set.add(sb.toString());
        }
        return set.size();
    }
}
