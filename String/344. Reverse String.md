class Solution {
    /**
     * Key point: swapping start and end
     *
     * Time Complexity: O(n)
     * Space Complexity: O(1) in place
     */
    public void reverseString(char[] s) {
        int start = 0, end = s.length - 1;
        char temp;
        while (start < end) {
            temp = s[start];
            s[start] = s[end];
            s[end] = temp;
            start++;
            end--;
        }
    }
}
