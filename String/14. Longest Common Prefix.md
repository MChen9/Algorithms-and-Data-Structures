class Solution {
    public String longestCommonPrefix(String[] strs) {
        int len = strs.length;
        if (len == 0) return "";
        StringBuilder pre = new StringBuilder(strs[0]);
        for (int i = 1; i < len; i++) {
            while (strs[i].indexOf(pre.toString()) != 0) {
                pre.deleteCharAt(pre.length() - 1);
            }
        }
        return pre.toString();
    }
}
