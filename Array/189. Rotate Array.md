class Solution {
    /**
     * 
     */
    public void rotate(int[] nums, int k) {
        int n = nums.length;
        k %= n;
        reverse(nums, 0, n - 1);
        reverse(nums, 0, k - 1);
        reverse(nums, k, n - 1);
    }
    
    private void reverse(int[] array, int s, int e) {
        int temp = 0;
        while (s < e) {
            temp = array[s];
            array[s] = array[e];
            array[e] = temp;
            s++;
            e--;
        }
    }
}
