class Solution {
    /**
     * Reverse is more easier on symmetric array than asymmetric one.
     * No need additional space.
     * After k rotation, the first k elements are the last k elements of 
     * original array, and the n - k elements are the first n - k elements
     * of the original. 
     * While the order should be reversed, we do the first reverse for 
     * moving the last k elements to the beginning and the first n - k
     * elements to the last, and we do the second and the third reverse
     * for change their order to the correct order.
     *
     * Time Complexity: O(n) about 2n times swap
     * Space Complexity: O(1) in place
     */
    public void rotate(int[] nums, int k) {
        int n = nums.length;
        k %= n;
        reverse(nums, 0, n - 1);
        reverse(nums, 0, k - 1);
        reverse(nums, k, n - 1);
    }
    
    private void reverse(int[] array, int s, int e) {
        int temp = 0;
        while (s < e) {
            temp = array[s];
            array[s] = array[e];
            array[e] = temp;
            s++;
            e--;
        }
    }
}
