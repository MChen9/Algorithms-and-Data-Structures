class Solution {
    public String getHint(String secret, String guess) {
        int bulls = 0, cows = 0;
        int[] nums = new int[10];
        char s, g;
        for (int i = 0; i < secret.length(); i++) {
            s = secret.charAt(i);
            g = guess.charAt(i);
            if (s == g) {
                bulls++;
            } else {
                if (nums[s - '0'] < 0) cows++;
                if (nums[g - '0'] > 0) cows++;
                    
                nums[s - '0']++;
                nums[g - '0']--;
            }
        }
        return bulls + "A" + cows + "B";
    }
}
