class Solution {
    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] R = new int[n];
        
        R[n-1] = nums[n-1];
        for (int i = n - 2; i >= 0; i--) {
            R[i] = R[i + 1] * nums[i];
        }
        int temp = 1;
        for (int i = 0; i < n; i++) {
            int right = (i + 1 < n)? R[i + 1]: 1;
            R[i] = temp * right;
            temp *= nums[i];
        }
        return R;
    }
}
