class Solution {
    /**
     */
    public int maxArea(int[] height) {
        int p1 = 0, p2 = height.length - 1;
        int maxArea = 0;
        int h1, h2;
        while (p1 < p2) {
            h1 = height[p1];
            h2 = height[p2];
            maxArea = Math.max(
                maxArea, Math.min(h1, h2) * (p2 - p1));
            if (h1 > h2)
                p2--;
            else
                p1++;
        }
        return maxArea;
    }
}
