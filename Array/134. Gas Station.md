class Solution {
    public int canCompleteCircuit(int[] gas, int[] cost) {
        int tank = 0, start = 0, sumGas = 0, sumCost = 0;
        for (int i = 0; i < gas.length; i++) {
            tank += gas[i] - cost[i];
            sumGas += gas[i];
            sumCost += cost[i];
            if (tank < 0) {
                start = i + 1;
                tank = 0;
            }
        }
        
        if (sumGas < sumCost)
            return -1;
        return start;
    }
}
