class Solution {
    public int removeDuplicates(int[] nums) {
        int n = nums.length;
        if (n < 3) return n;
        
        int p1 = 2, p2 = 2;
        while (p2 < n) {
            if (nums[p1 - 2] != nums[p2]) {
                nums[p1++] = nums[p2]; 
            }
            p2++;
        }
        return p1;
    }
}
