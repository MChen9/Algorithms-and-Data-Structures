class Solution {
    public double myPow(double x, int n) {
        double ans = 1;
        long N = Math.abs((long) n); // cast n to avoid losing digits
        while (N > 0) {
            if ((N & 1) == 1) // check ood or even
                ans *= x;
            x *= x;
            N >>= 1; // N is divided by 2
        }
        return (n > 0) ? ans: 1/ ans;
    }
}
