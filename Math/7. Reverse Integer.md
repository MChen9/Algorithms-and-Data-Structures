class Solution {
    public int reverse(int x) {         
        int revNum = 0, temp = 0;
        while (x != 0) {
            temp = temp * 10 + x % 10;
            if ((temp - x % 10) / 10 != revNum)
                return 0;
            revNum = temp;
            x /= 10;
        }
        
        return revNum;
    }
}
