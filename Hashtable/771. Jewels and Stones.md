class Solution {
    public int numJewelsInStones(String J, String S) {
        Set<Character> set = new HashSet<>();
        int cnt = 0;
        
        for (char j: J.toCharArray()) {
            set.add(j);
        }
        
        for (char s: S.toCharArray()) {
            if (set.contains(s))
                cnt++;
        }
        return cnt;
    }
}
